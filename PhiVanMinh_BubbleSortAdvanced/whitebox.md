### Code:

```
0	void bubbleSortAdvanced(int arraySort[], int n){
1		int isSort, changePlace;
2		do {
3			isSort = 0;
4			for (int i = 0; i < n-1; i++) {
5				if (arraySort[i] > arraySort[i+1]) {
6					changePlace = arraySort[i];
7					arraySort[i] = arraySort[i+1];
8					arraySort[i+1] = changePlace;
9					isSort = 1;
10				}
11			}
12		} while (isSort);
13	}
```
